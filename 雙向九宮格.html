void Show(uint16_t x)
{
		if(x==2||x==12||x==14||x==15){
		  PE0 = 1;
		}
		else{
		  PE0 = 0;
		}
		
		if(x==1||x==2||x==3||x==7||x==13){
		  PE2 = 1;
		}
		else{
		  PE2 = 0;
		}
		
		if(x==1||x==4||x==6||x==11||x==13){
		  PE3 = 1;
		}
		else{
		  PE3 = 0;
		}
		
		if(x==5||x==6||x==11||x==12||x==14||x==15){
		  PE4 = 1;
		}
		else{
		  PE4 = 0;
		}
		
		if(x==1||x==4||x==7||x==9||x==10||x==15){
		  PE5 = 1;
		}
		else{
		  PE5 = 0;
		}
		
		if(x==1||x==3||x==4||x==5||x==7||x==9){
		  PE6 = 1;
		}
		else{
		  PE6 = 0;
		}
		
		if(x==0||x==1||x==7||x==12){
		  PE7 = 1;
		}
		else{
		  PE7 = 0;
		}
}

void Display_7seg(int x)
{
	int a = 0, n = 0;
	int x1,x2,x3,x4;
	x1=x%10;
	x2=x/10%10;
	x3=x/10/10%10;
	x4=x/10/10/10;
	
	while(a<=200){
		PC4 = 1;
		PC5 = 0;
		PC6 = 0;
		PC7 = 0;
		Show(x1);
		CLK_SysTickDelay(500);
		PC4 = 0;
		PC5 = 1;
		PC6 = 0;
		PC7 = 0;
		Show(x2);
		CLK_SysTickDelay(500);
		PC4 = 0;
		PC5 = 0;
		PC6 = 1;
		PC7 = 0;
		Show(x3);
		CLK_SysTickDelay(500);
		PC4 = 0;
		PC5 = 0;
		PC6 = 0;
		PC7 = 1;
		Show(x4);
		CLK_SysTickDelay(500);
		a++;
	}
}

int sk(int i, int tem){
	while(i == 0){
		i = ScanKey();
	}
	tem = i;
	while(i != 0){
		i = ScanKey();
	}
	CLK_SysTickDelay(100000);
	return tem;
}

int winner(int i) {
	for(i = 0;i < 3;i++){
		if((ans[0][i] == ans[1][i])&&(ans[1][i] == ans[2][i])&&(ans[1][i] != 0)){
			return ans[0][i];
		}
		if((ans[i][0] == ans[i][1])&&(ans[i][1] == ans[i][2])&&(ans[i][1] != 0)){
			return ans[i][0];
		}
	}
	if((ans[0][0] == ans[1][1])&&(ans[1][1] == ans[2][2])){
		return ans[0][0];
	}
	if((ans[0][2] == ans[1][1])&&(ans[1][1] == ans[2][0])){
		return ans[0][2];
	}
	return 0;
}
